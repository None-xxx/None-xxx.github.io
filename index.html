#include<bits/stdc++.h>
using namespace std;
int N;
int n, m, k;
char mm[202][202];
int s[32][202][202];
int qh[11], ql[11]; 
int sh, sl, eh, el;
int c;
int dh[4] = {0, 0, -1, 1};
int dll[4] = {1, -1, 0, 0};
struct p
{
	int h;
	int l;
	int key;
};
int main()
{
	cin >> N;
	for(int o = 1; o <= N; o ++)
	{
		cin >> n >> m >> k; c=0; 
		memset(mm, 0, sizeof(mm));
		memset(qh, 0, sizeof(qh));
		memset(ql, 0, sizeof(ql));
		memset(s, -1, sizeof(s));
		queue <p> dl;
		for(int i = 1; i <= n; i ++)
		{
			for(int j = 1; j <= m; j ++)
			{
				cin >> mm[i][j];
				if(mm[i][j] == 'S') 
				{
					s[0][i][j] = 0;
					p pp;
					pp.h = i;
					pp.l = j;
					pp.key = 0;
					dl.push(pp);
				}
				if(mm[i][j] == 'E')
				{
					eh = i;
					el = j;
				}
				if(mm[i][j] == '$')
				{
					c ++;
					qh[c] = i;
					ql[c] = j;
				}
			}
		}
		while(1)
		{
			if(dl.empty())
			{
				break;
			}
			p a = dl.front();
			int h = a.h;
			int l = a.l;
			int key = a.key;
			//cout << "h:" << h << "   l:" << l << "   key:" << key << endl; 
			for(int i = 0; i <= 3; i ++)
			{
				int nh = h + dh[i];
				int nl = l + dll[i];
				int nkey = key;
				if(mm[nh][nl] == '.' || mm[nh][nl] == 'E')
				{
					if(s[nkey][nh][nl] == -1)
					{
						s[nkey][nh][nl] = s[key][h][l] + 1;
						p tmp;
						tmp.h = nh;
						tmp.l = nl;
						tmp.key = nkey;
						dl.push(tmp);
					}
				}
				if(mm[nh][nl] == '$')
				{
					for(int j = 1; j <= c; j ++)
					{
						if(s[nkey][qh[j]][ql[j]] == -1)
						{
							s[nkey][qh[j]][ql[j]] = s[key][h][l] + 1;
							p tmp;
							tmp.h = qh[j];
							tmp.l = ql[j];
							tmp.key = nkey;
							dl.push(tmp);
						}
					}
				}
				if(mm[nh][nl] <= '9' && mm[nh][nl] >= '0')
				{
					nkey = nkey | (1<<(mm[nh][nl]-'0'));
					if(s[nkey][nh][nl] == -1)
					{
						s[nkey][nh][nl] = s[key][h][l] + 1;
						p tmp;
						tmp.h = nh;
						tmp.l = nl;
						tmp.key = nkey;
						dl.push(tmp);
					}
				}
			}
			dl.pop();
		}
		//cout<<(1<<(k))-1<<" "<<eh<<" "<<el<<endl;
		if(s[(1<<(k))-1][eh][el] != -1)
		{
			cout << s[(1<<(k))-1][eh][el] << endl;
		}
		else
		{
			cout << "oop!" << endl;
		}
	}	
	return 0;
}
